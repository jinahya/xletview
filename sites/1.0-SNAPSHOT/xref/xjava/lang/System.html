140      * System properties. The following properties are guaranteed to be defined:
141      * <dl>
142      * <dt>java.version     <dd>Java version number
143      * <dt>java.vendor      <dd>Java vendor specific string
144      * <dt>java.vendor.url  <dd>Java vendor URL
145      * <dt>java.home        <dd>Java installation directory
146      * <dt>java.class.version   <dd>Java class version number
147      * <dt>java.class.path  <dd>Java classpath
148      * <dt>os.name      <dd>Operating System Name
149      * <dt>os.arch      <dd>Operating System Architecture
150      * <dt>os.version       <dd>Operating System Version
151      * <dt>file.separator   <dd>File separator ("/" on Unix)
152      * <dt>path.separator   <dd>Path separator (":" on Unix)
153      * <dt>line.separator   <dd>Line separator ("\n" on Unix)
154      * <dt>user.name        <dd>User account name
155      * <dt>user.home        <dd>User home directory
156      * <dt>user.dir     <dd>User's current working directory
157      * </dl>

161     private static Properties initProperties(Properties props) {
162         Properties real = java.lang.System.getProperties();
163         props.setProperty("java.version", real.getProperty("java.version"));
164         props.setProperty("java.vendor", real.getProperty("java.vendor"));
165         
166         /*
167          * this one is not guaranteed to exist but I put it
168          * here because there was applications at a broadcaster that use
169          * it.
170          */  
171         props.setProperty("java.vm.vendor", real.getProperty("java.vm.vendor"));
172         
173         props.setProperty("java.vendor.url", real.getProperty("java.vendor.url"));
174         props.setProperty("java.home", real.getProperty("java.home"));
175         props.setProperty("java.class.version", real.getProperty("java.class.version"));
176         props.setProperty("java.class.path", real.getProperty("java.class.path"));
177         props.setProperty("os.name", real.getProperty("os.name"));
178         props.setProperty("os.arch", real.getProperty("os.arch"));
179         props.setProperty("os.version", real.getProperty("os.version"));
180         props.setProperty("file.separator", real.getProperty("file.separator"));
181         props.setProperty("path.separator", real.getProperty("path.separator"));
182         props.setProperty("line.separator", real.getProperty("line.separator"));
183         props.setProperty("user.name", real.getProperty("user.name"));
184         props.setProperty("user.home", real.getProperty("user.home"));
185         props.setProperty("user.dir", real.getProperty("user.dir"));
186         //props.list(java.lang.System.out);    
187         return props;
188     }

190     public static Properties getProperties() {

194     public static void setProperties(Properties props) {
195         if (props == null) {
196             props = new Properties();
197             initProperties(props);
198         }
199         System.props = props;
200     }

202     public static String getProperty(String key) {
203         if (key == null) {
204             throw new NullPointerException("key can't be null");
205         }
206         if (key.equals("")) {
207             throw new IllegalArgumentException("key can't be empty");
208         }
209         return props.getProperty(key);
210     }
211 
212     public static String getProperty(String key, String def) {
213         if (key == null) {
214             throw new NullPointerException("key can't be null");
215         }
216         if (key.equals("")) {
217             throw new IllegalArgumentException("key can't be empty");
218         }
219         return props.getProperty(key, def);
220     }
221 
222     public static String setProperty(String key, String value) {
223         if (key == null) {
224             throw new NullPointerException("key can't be null");
225         }
226         if (key.equals("")) {
227             throw new IllegalArgumentException("key can't be empty");
228         }
229         return (String) props.setProperty(key, value);
230     }
