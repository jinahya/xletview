14  package xjavax.tv.media;
15  
16  /**
17   * <code>AWTVideoSizeControl</code> allows setting clipping, scaling, and
18   * translation of a video stream in a simple, interoperable way.  Not
19   * all possible combinations of positioning will be supported, so this
20   * interface provides a mechanism to discover how closely the
21   * underlying platform will approximate a request for positioning.
22   * 
23   * <p> All interactions via AWTVideoSizeControl happen in the
24   * coordinate space of the screen.  For example, successfully setting
25   * the video's position to the location reported by
26   * <code>Component.getLocationOnScreen()</code> on the
27   * <code>Xlet</code>'s root container will cause the upper left-hand
28   * corner of the video and the root container to coincide.  <p> The
29   * screen, in the context of AWT, is the area into which graphics
30   * drawing operations are done.  Its size is given by
31   * java.awt.Toolkit.getScreenSize(), and locations reported by
32   * Component.getLocationOnScreen() are given in the screen's
33   * coordinate system.<p>
34   * 
35   * Instances of <code>AWTVideoSizeControl</code> may be obtained from
36   * a JMF <code>Player</code> via the methods
37   * <code>getControl(String)</code> and <code>getControls()</code>.
38   * Note that a Java TV API implementation may not always or ever
39   * support <code>AWTVideoSizeControl</code> for a given Player; in
40   * such a case, the failure modes specified by the two aforementioned
41   * methods will apply.
42   * <CODE>java.awt.Component.getLocationOnScreen()</CODE>,
43   * <CODE>Player</CODE></DL>
44   * <HR>
45   * 
46   * 
47   */
48  public interface AWTVideoSizeControl extends javax.media.Control
49  {
50  	/**
51  	 * Reports the <code>AWTVideoSize</code> at which the Player is
52  	 * currently operating.
53  	 * 
54  	 * @return A copy of the JMF Player's current video size, in the AWT coordinate space.
55  	 */
56  	public AWTVideoSize getSize();
57  
58  	/**
59  	 * Reports the default <code>AWTVideoSize</code> for this control.
60  	 * For the background video plane, this will be the size that the
61  	 * video would be presented at if no program had manipulated the
62  	 * video size.
63  	 * 
64  	 * @return The default AWTVideoSize.
65  	 */
66  	public AWTVideoSize getDefaultSize();
67  
68  	/**
69  	 * Reports the size of the source video, in the screen's
70  	 * coordinate system.
71  	 * 
72  	 * @return The size of the source video.
73  	 */
74  	public java.awt.Dimension getSourceVideoSize();
75  
76  	/**
77  	 * Sets the video size.  If the size provided cannot be supported
78  	 * by the underlying platform, this method does nothing and
79  	 * returns <code>false</code>.
80  	 * 
81  	 * @param sz - The desired video size, in the AWT coordinate space.
82  	 * @return true if the size was successfully changed; false if the platform is incapable of supporting the given size.
83  	 * @see checkSize(AWTVideoSize)
84  	 */
85  	public boolean setSize( AWTVideoSize sz);
86  
87  	/**
88  	 * Reports how closely the underlying platform can approximate a
89  	 * desired video size.  If the underlying platform cannot support
90  	 * the given size, this method gives the closest approximation
91  	 * that the platform is capable of.
92  	 * 
93  	 * @param sz - The desired video size.
94  	 * @return The actual size that the platform would be able to set.
95  	 */
96  	public AWTVideoSize checkSize( AWTVideoSize sz);
97  
98  }
