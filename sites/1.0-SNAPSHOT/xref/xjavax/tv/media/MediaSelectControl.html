14  package xjavax.tv.media;
15  
16  import xjavax.tv.locator.InvalidLocatorException;
17  import xjavax.tv.locator.Locator;
18  import xjavax.tv.service.selection.InsufficientResourcesException;
19  import xjavax.tv.service.selection.InvalidServiceComponentException;
20  
21  /**
22   * <code>MediaSelectControl</code> allows the selection of different
23   * kinds of content in a running <code>Player</code>.  It serves as
24   * a high level demultiplex control, where the selection is specified
25   * by locators indicating one or more service components to
26   * present. <p>
27   * 
28   * If the <code>Player</code> on which a
29   * <code>MediaSelectControl</code> operates is an instance of
30   * <code>ServiceMediaHandler</code>, then
31   * <code>MediaSelectControl</code> is restricted to operating only on
32   * service components belonging to the service with which the
33   * <code>ServiceMediaHandler</code> is associated (that is, the
34   * <code>ServiceContext</code>'s currently selected service).<p>
35   * 
36   * Instances of <code>MediaSelectControl</code> may be obtained from a
37   * JMF <code>Player</code> via the methods
38   * <code>getControl(String)</code> and <code>getControls()</code>.
39   * Note that a Java TV API implementation may not always or ever
40   * support <code>MediaSelectControl</code> for a given Player; in such
41   * a case, the failure modes specified by the two aforementioned
42   * methods will apply.
43   * <CODE>Controller.getControls()</CODE>,
44   * <A HREF="../../../javax/tv/service/selection/ServiceMediaHandler.html"><CODE>ServiceMediaHandler</CODE></A>,
45   * <A HREF="../../../javax/tv/service/selection/ServiceContext.html"><CODE>ServiceContext</CODE></A></DL>
46   * <HR>
47   * 
48   * 
49   */
50  public interface MediaSelectControl extends javax.media.Control
51  {
52  	/**
53  	 * Selects a new service component for presentation.  If some
54  	 * content is currently playing, it is replaced in its entirety by
55  	 * the specified selection.  This is an asynchronous operation
56  	 * that is completed upon receipt of a
57  	 * <code>MediaSelectEvent</code>.  Note that for certain
58  	 * selections that imply a different time base or otherwise change
59  	 * synchronization relationships, a <code>RestartingEvent</code>
60  	 * will be posted by the <code>Player</code>.
61  	 * 
62  	 * @param component - A locator representing an individual service component to present.
63  	 * @throws InvalidLocatorException - If the locator does not reference a selectable service component.
64  	 * @throws InvalidServiceComponentException - If the specified service component is not part of the Service to which the MediaSelectControl is restricted, or if it cannot be presented alone.
65  	 * @throws InsufficientResourcesException - If the operation cannot be completed due to a lack of system resources.
66  	 * @throws java.lang.SecurityException - If the caller does not have MediaSelectPermission(component) permission.
67  	 */
68  	public void select( Locator component) throws InvalidLocatorException, InvalidServiceComponentException, InsufficientResourcesException, java.lang.SecurityException;
69  
70  	/**
71  	 * Selects one or more service components for presentation.  If
72  	 * some content is currently playing, it is replaced in its
73  	 * entirety by the specified selection.  This is an asynchronous
74  	 * operation that is completed on receipt of a
75  	 * <code>MediaSelectEvent</code>.  Note that for certain
76  	 * selections that imply a different time base or otherwise change
77  	 * synchronization relationships, a <code>RestartingEvent</code>
78  	 * will be posted by the <code>Player</code>.
79  	 * 
80  	 * @param components - An array of locators representing a set of individual service components to present together.
81  	 * @throws InvalidLocatorException - If a locator provided does not reference a selectable service component.
82  	 * @throws InvalidServiceComponentException - If a specified service component is not part of the Service to which the MediaSelectControl is restricted, if a specified service component must be presented in conjunction with another service component not contained in components, if the specified set of service components cannot be presented as a coherent whole, or if the service components are not all available simultaneously.
83  	 * @throws InsufficientResourcesException - If the operation cannot be completed due to a lack of system resources.
84  	 * @throws java.lang.SecurityException - If the caller does not have MediaSelectPermission(components[i]) permission for any valid i.
85  	 */
86  	public void select( Locator[] components) throws InvalidLocatorException, InvalidServiceComponentException, InsufficientResourcesException, java.lang.SecurityException;
87  
88  	/**
89  	 * Adds a service component (for example, subtitles) to the
90  	 * presentation. This is an asynchronous operation that is
91  	 * completed on receipt of a <code>MediaSelectEvent</code>.
92  	 * Components whose addition would require Player
93  	 * resynchronization are not permitted.  If the specified service
94  	 * component is already part of the presentation, this method does
95  	 * nothing.
96  	 * 
97  	 * @param component - The locator representing an individual service component to add to the presentation.
98  	 * @throws InvalidLocatorException - If the specified locator does not reference a selectable service component.
99  	 * @throws InvalidServiceComponentException - If the addition of the service component would require resynchronization of the Player, if the service component is not part of the Service to which the MediaSelectControl is restricted, or if the service component must be presented in conjunction with another service component that is not part of the current presentation.
100 	 * @throws InsufficientResourcesException - If the operation cannot be completed due to a lack of system resources.
101 	 * @throws java.lang.SecurityException - If the caller does not have MediaSelectPermission(component) permission.
102 	 */
103 	public void add( Locator component) throws InvalidLocatorException, InvalidServiceComponentException, InsufficientResourcesException, java.lang.SecurityException;
104 
105 	/**
106 	 * Removes a service component from the presentation. This is an
107 	 * asynchronous operation that is completed on receipt of a
108 	 * <code>MediaSelectEvent</code>. Components whose removal would
109 	 * require Player resynchronization are not permitted.
110 	 * 
111 	 * @param component - The locator representing an individual service component to remove from the presentation.
112 	 * @throws InvalidLocatorException - If the specified locator does not reference a service component in the current selection.
113 	 * @throws InvalidServiceComponentException - If removal of the specified service component would require resynchronization of the Player, or if another service component in the current presentation must be presented in conjunction with component.
114 	 * @throws java.lang.SecurityException - If the caller does not have MediaSelectPermission(component) permission.
115 	 */
116 	public void remove( Locator component) throws InvalidLocatorException, InvalidServiceComponentException, java.lang.SecurityException;
117 
118 	/**
119 	 * Replaces a service component in the presentation. This is an
120 	 * asynchronous operation that is completed on receipt of a
121 	 * <code>MediaSelectEvent</code>. Components whose replacement
122 	 * would require Player resynchronization are not permitted.
123 	 * 
124 	 * @param fromComponent - The locator that represents the service component to remove from the presentation.
125 	 * @param toComponent - The locator that represents the service component to add to the presentation.
126 	 * @throws InvalidLocatorException - If fromComponent does not reference a service component in the current selection, or if toComponent does not reference a selectable service component.
127 	 * @throws InvalidServiceComponentException - If toComponent references a service component that is not part of the Service to which the MediaSelectControl is restricted, if fromComponent or toComponent reference service components for which this operation would require resynchronization of the Player, if another service component in the current presentation must be presented in conjunction with fromComponent, or if toComponent must be presented in conjunction with a service component not in the resulting presentation.
128 	 * @throws InsufficientResourcesException - If the operation cannot be completed due to a lack of system resources.
129 	 * @throws java.lang.SecurityException - If the caller does not have MediaSelectPermission(fromComponent) and MediaSelectPermission(toComponent) permission.
130 	 */
131 	public void replace( Locator fromComponent, Locator toComponent) throws InvalidLocatorException, InvalidServiceComponentException, InsufficientResourcesException, java.lang.SecurityException;
132 
133 	/**
134 	 * Subscribes the specified <code>MediaSelectListener</code> to
135 	 * receive events related to media selection on this Player.
136 	 * 
137 	 * @param listener - The MediaSelectListener to which to send events.
138 	 */
139 	public void addMediaSelectListener( MediaSelectListener listener);
140 
141 	/**
142 	 * Unsubscribes the specified <code>MediaSelectListener</code> from
143 	 * receiving events related to media selection on this Player.
144 	 * 
145 	 * @param listener - The MediaSelectListener to unsubscribe.
146 	 */
147 	public void removeMediaSelectListener( MediaSelectListener listener);
148 
149 	/**
150 	 * Reports the components of the current selection.
151 	 * 
152 	 * @return An array of locators representing the service components in the current selection.
153 	 */
154 	public Locator[] getCurrentSelection();
155 
156 }
