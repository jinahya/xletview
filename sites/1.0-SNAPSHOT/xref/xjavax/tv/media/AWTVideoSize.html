14  package xjavax.tv.media;
15  
16  /**
17   * <code>AWTVideoSize</code> is a data holder that represents the position,
18   * scaling, and clipping of a JMF Player, as controlled via an
19   * AWTVideoSizeControl. All coordinates are expressed in the same coordinate
20   * space as AWT components. Because background video might be larger than the
21   * addressable AWT area, some of the positions might be negative.
22   * 
23   * <p>
24   * An AWTVideoSize represents a transformation of video where the video is
25   * first positioned, then scaled, and then clipped. A rectangle (in the
26   * screen's coordinate system) of the source video is translated, scaled and
27   * clipped to fit within a rectangle specified in the screen's coordinate
28   * system.
29   * <HR>
30   * 
31   * @author Martin Sveden
32   * @statuscode 4
33   */
34  public class AWTVideoSize {
35  	private java.awt.Rectangle source;
36  	private java.awt.Rectangle destination;
37  	private float xScale;
38  	private float yScale;
39  
40  	/**
41  	 * Constructs a new <code>AWTVideoSize</code> instance. This <code>AWTVideoSize</code>
42  	 * represents a transformation where the rectangle <code>source</code> in
43  	 * the source video is scaled and clipped to be within the rectangle <code>dest</code>.
44  	 * 
45  	 * <p>
46  	 * The instance of AWTVideoSize created with this constructor will not
47  	 * maintain a reference to either of the constructor's parameters.
48  	 * 
49  	 * @param source -
50  	 *            The rectangle representing the portion of the source video to
51  	 *            display, in the coordinate system of the screen.
52  	 * @param dest -
53  	 *            The rectangle representing where the video is to be
54  	 *            displayed, in the coordinate system of the screen.
55  	 */
56  	public AWTVideoSize(java.awt.Rectangle source, java.awt.Rectangle dest) {
57  		this.source = source;
58  		this.destination = dest;
59  		try{
60  			xScale = destination.width / source.width;
61  			yScale = destination.height / source.height;			
62  		}
63  		catch(ArithmeticException e){
64  			e.printStackTrace();
65  		}
66  
67  	}
68  
69  	/**
70  	 * Return a copy of the rectangle representing the portion of the source
71  	 * video to display, in the coordinate system of the screen.
72  	 * 
73  	 * @return The source Rectangle.
74  	 */
75  	public java.awt.Rectangle getSource() {
76  		return this.source;
77  	}
78  
79  	/**
80  	 * Return a copy of the rectangle representing where the video is to be
81  	 * displayed, in the coordinate system of the screen.
82  	 * 
83  	 * @return The destination Rectangle.
84  	 */
85  	public java.awt.Rectangle getDestination() {
86  		return this.destination;
87  	}
88  
89  	/**
90  	 * Give the scaling factor applied to the video in the horizontal
91  	 * dimension, i.e., <code>getDestination().width / getSource().width</code>.
92  	 * 
93  	 * @return The horizontal scaling factor applied to the video.
94  	 */
95  	public float getXScale() {
96  		return this.xScale;
97  	}
98  
99  	/**
100 	 * Give the scaling factor applied to the video in the vertical dimension,
101 	 * i.e., <code>getDestination().height / getSource().height</code>.
102 	 * 
103 	 * @return The vertical scaling factor applied to the video.
104 	 */
105 	public float getYScale() {
106 		return this.yScale;
107 	}
108 
109 	/**
110 	 * Generates a hash code value for this <code>AWTVideoSize</code>. Two
111 	 * <code>AWTVideoSize</code> instances for which <code>AWTVideoSize.equals()</code>
112 	 * is <code>true</code> will have identical hash code values.
113 	 * 
114 	 * @return The hashcode value for this AWTVideoSize.
115 	 * @see hashCode in class java.lang.Object
116 	 */
117 	public int hashCode() {
118 		return toString().hashCode();
119 	}
120 
121 	/**
122 	 * Compares this <code>AWTVideoSize</code> with the given object for
123 	 * equality. Returns <code>true</code> if and only if the given object is
124 	 * also of type <code>AWTVideoSize</code> and contains data members equal
125 	 * to those of this <code>AWTVideoSize</code>.
126 	 * 
127 	 * @param other -
128 	 *            The object with which to test for equality.
129 	 * @return true if the two AWTVideoSize instances are equal; false
130 	 *         otherwise.
131 	 * @see equals in class java.lang.Object
132 	 */
133 	public boolean equals(java.lang.Object other) {
134 		boolean result = false;
135 		
136 		if(other instanceof AWTVideoSize){
137 			if(this.hashCode() == other.hashCode()){
138 				result = true;
139 			}
140 		}
141 		
142 		return result;
143 	}
144 
145 	/**
146 	 * Returns a string representation of this <code>AWTVideoSize</code> and
147 	 * its values.
148 	 * 
149 	 * @return A string representation of this object.
150 	 * @see toString in class java.lang.Object
151 	 */
152 	public java.lang.String toString() {
153 		String theSource = "source[x=" + source.x + ", y=" + source.y + ", width=" + source.width + ", height=" + source.height + "]"; 
154 		String theDestination = "destination[x=" + destination.x + ", y=" + destination.y + ", width=" + destination.width + ", height=" + destination.height + "]";
155 		String theXScale = "xScale=" + xScale;
156 		String theYScale = "yScale=" + yScale;
157 		String result = theSource + ", " + theDestination + ", " + theXScale + ", " + theYScale;		
158 		return result;
159 	}
160 
161 }
