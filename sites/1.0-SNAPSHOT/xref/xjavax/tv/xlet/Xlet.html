14  package xjavax.tv.xlet;
15  
16  /**
17   * This interface allows an application manager to create,
18   * initialize, start, pause, and destroy an Xlet.
19   * An Xlet is an application or service designed to be run and
20   * controlled by an application manager via this lifecycle interface.
21   * The lifecycle states allow the application manager to manage
22   * the activities of multiple Xlets within a runtime environment
23   * selecting which Xlets are active at a given time.
24   * The application manager maintains the state of the Xlet and
25   * invokes method on the Xlet via the lifecycle methods.  The Xlet
26   * implements these methods to update its internal activities and
27   * resource usage as directed by the application manager.
28   * The Xlet can initiate some state changes itself and informs
29   * the application manager of those state changes
30   * by invoking methods on <code>XletContext</code>.<p>
31   * 
32   * In order to support interoperability between Xlets and application
33   * managers, all Xlet classes must provide a public no-argument
34   * constructor.<p>
35   * 
36   * <b>Note:</b> The methods on this interface are meant to signal state
37   * changes. The state change is not considered complete until the state
38   * change method has returned. It is intended that these methods return
39   * quickly.<p>
40   * <HR>
41   * 
42   * 
43   */
44  public interface Xlet
45  {
46  	/**
47  	 * Signals the Xlet to initialize itself and enter the
48  	 * <i>Paused</i> state.
49  	 * The Xlet shall initialize itself in preparation for providing service.
50  	 * It should not hold shared resources but should be prepared to provide
51  	 * service in a reasonable amount of time. <p>
52  	 * An <code>XletContext</code> is used by the Xlet to access
53  	 * properties associated with its runtime environment.
54  	 * After this method returns successfully, the Xlet
55  	 * is in the <i>Paused</i> state and should be quiescent. <p>
56  	 * <b>Note:</b> This method shall only be called once.<p>
57  	 * 
58  	 * @param ctx - The XletContext of this Xlet.
59  	 * @throws XletStateChangeException - If the Xlet cannot be initialized.
60  	 * @see XletContext
61  	 */
62  	public void initXlet( XletContext ctx) throws XletStateChangeException;
63  
64  	/**
65  	 * Signals the Xlet to start providing service and
66  	 * enter the <i>Active</i> state.
67  	 * In the <i>Active</I> state the Xlet may hold shared resources.
68  	 * The method will only be called when
69  	 * the Xlet is in the <i>paused</i> state.
70  	 * <p>
71  	 * 
72  	 * @throws XletStateChangeException - is thrown if the Xlet cannot start providing service.
73  	 */
74  	public void startXlet() throws XletStateChangeException;
75  
76  	/**
77  	 * Signals the Xlet to stop providing service and
78  	 * enter the <i>Paused</i> state.
79  	 * In the <i>Paused</i> state the Xlet must stop providing
80  	 * service, and might release all shared resources
81  	 * and become quiescent. This method will only be called
82  	 * called when the Xlet is in the <i>Active</i> state. <p></DL>
83  	 * 
84  	 */
85  	public void pauseXlet();
86  
87  	/**
88  	 * Signals the Xlet to terminate and enter the <i>Destroyed</i> state.
89  	 * In the destroyed state the Xlet must release
90  	 * all resources and save any persistent state. This method may
91  	 * be called from the <i>Loaded</i>, <i>Paused</i> or
92  	 * <i>Active</i> states. <p>
93  	 * Xlets should
94  	 * perform any operations required before being terminated, such as
95  	 * releasing resources or saving preferences or
96  	 * state. <p>
97  	 * 
98  	 * <b>NOTE:</b> The Xlet can request that it not enter the <i>Destroyed</i>
99  	 * state by throwing an <code>XletStateChangeException</code>. This
100 	 * is only a valid response if the <code>unconditional</code>
101 	 * flag is set to <code>false</code>. If it is <code>true</code>
102 	 * the Xlet is assumed to be in the <i>Destroyed</i> state
103 	 * regardless of how this method terminates. If it is not an
104 	 * unconditional request, the Xlet can signify that it wishes
105 	 * to stay in its current state by throwing the Exception.
106 	 * This request may be honored and the <code>destroyXlet()</code>
107 	 * method called again at a later time.
108 	 * 
109 	 * @param unconditional - If unconditional is true when this method is called, requests by the Xlet to not enter the destroyed state will be ignored.
110 	 * @throws XletStateChangeException - is thrown if the Xlet wishes to continue to execute (Not enter the Destroyed state). This exception is ignored if unconditional is equal to true.
111 	 */
112 	public void destroyXlet(boolean unconditional) throws XletStateChangeException;
113 
114 }
