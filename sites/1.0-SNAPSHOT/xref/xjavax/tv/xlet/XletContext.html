14  package xjavax.tv.xlet;
15  
16  /**
17   * An interface that provides methods allowing an Xlet to discover
18   * information about its environment. An XletContext is passed
19   * to an Xlet when the Xlet is initialized. It provides an Xlet with a
20   * mechanism to retrieve properties, as well as a way to signal
21   * internal state changes.
22   * <A HREF="../../../javax/tv/graphics/TVContainer.html"><CODE>TVContainer</CODE></A></DL>
23   * <HR>
24   * 
25   * 
26   */
27  public interface XletContext
28  {
29  	/**
30  	 * The property key used to obtain initialization arguments for the
31  	 * Xlet.  The call
32  	 * <code>XletContext.getXletProperty(XletContext.ARGS)</code> will
33  	 * return the arguments as an array of Strings.  If there are
34  	 * no arguments, then an array of length 0 will be returned.
35  	 * <DT><B>See Also: </B>
36  	 * 
37  	 * 
38  	 */
39  	public static final java.lang.String ARGS = "javax.tv.xlet.args";
40  	
41  	
42  
43  	/**
44  	 * Used by an application to notify its manager that it
45  	 * has entered into the
46  	 * <i>Destroyed</i> state.  The application manager will not
47  	 * call the Xlet's <code>destroy</code> method, and all resources
48  	 * held by the Xlet will be considered eligible for reclamation.
49  	 * Before calling this method,
50  	 * the Xlet must have performed the same operations
51  	 * (clean up, releasing of resources etc.) it would have if the
52  	 * <code>Xlet.destroyXlet()</code> had been called.</DL>
53  	 * 
54  	 */
55  	public void notifyDestroyed();
56  
57  	/**
58  	 * Notifies the manager that the Xlet does not want to be active and has
59  	 * entered the <i>Paused</i> state.  Invoking this method will
60  	 * have no effect if the Xlet is destroyed, or if it has not
61  	 * yet been started. <p>
62  	 * 
63  	 * If an Xlet calls <code>notifyPaused()</code>, in the
64  	 * future it may receive an <i>Xlet.startXlet()</i> call to request
65  	 * it to become active, or an <i>Xlet.destroyXlet()</i> call to request
66  	 * it to destroy itself.</DL>
67  	 * 
68  	 */
69  	public void notifyPaused();
70  
71  	/**
72  	 * Provides an Xlet with a mechanism to retrieve named
73  	 * properties from the XletContext.
74  	 * 
75  	 * @param key - The name of the property.
76  	 * @return A reference to an object representing the property. null is returned if no value is available for key.
77  	 */
78  	public java.lang.Object getXletProperty(java.lang.String key);
79  
80  	/**
81  	 * Provides the Xlet with a mechanism to indicate that it is
82  	 * interested in entering the <i>Active</i> state. Calls to this
83  	 * method can be used by an application manager to determine which
84  	 * Xlets to move to <i>Active</i> state.  Any subsequent call to
85  	 * <code>Xlet.startXlet()</code> as a result of this method will
86  	 * be made via a different thread than the one used to call
87  	 * <code>resumeRequest()</code>.
88  	 * 
89  	 * @see Xlet.startXlet()
90  	 */
91  	public void resumeRequest();
92  
93  }
