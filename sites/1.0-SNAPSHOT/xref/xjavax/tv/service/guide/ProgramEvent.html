14  package xjavax.tv.service.guide;
15  
16  import xjavax.tv.service.SIElement;
17  import xjavax.tv.service.SIRequest;
18  import xjavax.tv.service.SIRequestor;
19  import xjavax.tv.service.Service;
20  
21  /**
22   * <code>ProgramEvent</code> represents collection of elementary
23   * streams with a common time base, an associated start time, and an
24   * associated end time. An event is equivalent to the common industry
25   * usage of "TV program." <p>
26   * 
27   * 
28   * The Event Information Table (EIT) contains information (titles, start
29   * times, etc.) for events on defined services. An event is, in most cases,
30   * a typical TV program, however its definition may be extended to include
31   * particular data broadcasting sessions and other information segments.<p>
32   * 
33   * A <code>ProgramEvent</code> object may optionally implement the
34   * <code>CAIdentification</code> interface. Note that all time values
35   * are in UTC time. <P>
36   * <A HREF="../../../../javax/tv/service/navigation/CAIdentification.html"><CODE>CAIdentification</CODE></A>,
37   * <a href="../../../../overview-summary.html#guidelines-opinterfaces">Optionally implemented interfaces</a></DL>
38   * <HR>
39   * 
40   * 
41   */
42  public interface ProgramEvent extends SIElement
43  {
44  	/**
45  	 * Returns the start time of this program event. The start time is in UTC
46  	 * time.
47  	 * 
48  	 * @return This program's start time (UTC).
49  	 */
50  	public java.util.Date getStartTime();
51  
52  	/**
53  	 * Returns the end time of this program event. The end time is in UTC time.
54  	 * 
55  	 * @return This program's end time (UTC).
56  	 */
57  	public java.util.Date getEndTime();
58  
59  	/**
60  	 * Returns the duration of this program event in seconds.
61  	 * 
62  	 * @return This program's duration in seconds.
63  	 */
64  	public long getDuration();
65  
66  	/**
67  	 * Returns the program event title. This information may be obtained in
68  	 * the ATSC EIT table or the DVB Short Event Descriptor.
69  	 * 
70  	 * @return A string representing this program's title, or an empty string if the title is unavailable.
71  	 */
72  	public java.lang.String getName();
73  
74  	/**
75  	 * Retrieves a textual description of the event. This method
76  	 * delivers its results asynchronously.
77  	 * 
78  	 * @param requestor - The SIRequestor to be notified when this retrieval operation completes.
79  	 * @return An SIRequest object identifying this asynchronous retrieval request.
80  	 * @see ProgramEventDescription
81  	 */
82  	public SIRequest retrieveDescription( SIRequestor requestor);
83  
84  	/**
85  	 * Reports content advisory information associated with this program for
86  	 * the local rating region.
87  	 * 
88  	 * @return A ContentRatingAdvisory object describing the rating of this ProgramEvent or null if no rating information is available.
89  	 */
90  	public ContentRatingAdvisory getRating();
91  
92  	/**
93  	 * Reports the <code>Service</code> this program event is associated with.
94  	 * 
95  	 * @return The Service this program event is delivered on.
96  	 */
97  	public Service getService();
98  
99  	/**
100 	 * Retrieves an array of service components which are part of this
101 	 * <code>ProgramEvent</code>.  Service component information may not
102 	 * always be available. If the <code>ProgramEvent</code> is current,
103 	 * this method will provide only service components associated with
104 	 * the <code>Service</code> to which the <code>ProgramEvent</code>
105 	 * belongs.  If the <code>ProgramEvent</code> is not current, no
106 	 * guarantee is provided that all or even any of its service
107 	 * components will be available.<p>
108 	 * 
109 	 * This method delivers its results asynchronously.  The retrieved
110 	 * array will only contain <code>ServiceComponent</code> instances
111 	 * <code>c</code> for which the caller has
112 	 * <code>javax.tv.service.ReadPermission(c.getLocator())</code>.  If
113 	 * no <code>ServiceComponent</code> instances meet this criteria,
114 	 * this method will result in an <code>SIRequestFailureType</code> of
115 	 * <code>DATA_UNAVAILABLE</code>.
116 	 * 
117 	 * @param requestor - The SIRequestor to be notified when this retrieval operation completes.
118 	 * @return An SIRequest object identifying this asynchronous retrieval request.
119 	 * @see ServiceComponent, ReadPermission
120 	 */
121 	public SIRequest retrieveComponents( SIRequestor requestor);
122 
123 }
