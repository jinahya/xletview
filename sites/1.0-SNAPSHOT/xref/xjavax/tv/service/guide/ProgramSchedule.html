14  package xjavax.tv.service.guide;
15  
16  import xjavax.tv.locator.InvalidLocatorException;
17  import xjavax.tv.locator.Locator;
18  import xjavax.tv.service.SIException;
19  import xjavax.tv.service.SIRequest;
20  import xjavax.tv.service.SIRequestor;
21  
22  /**
23   * This interface represents a collection of program events for a given
24   * service ordered by time. It provides the current, next and future
25   * program events.<p>
26   * 
27   * Note that all time values are in UTC time.
28   * <A HREF="../../../../javax/tv/service/guide/ProgramEvent.html"><CODE>ProgramEvent</CODE></A>,
29   * <A HREF="../../../../javax/tv/service/ReadPermission.html"><CODE>ReadPermission</CODE></A></DL>
30   * <HR>
31   * 
32   * 
33   */
34  public interface ProgramSchedule
35  {
36  	/**
37  	 * Retrieves the current <code>ProgramEvent</code>.  The resulting
38  	 * <code>ProgramEvent</code> is available for immediate viewing.<p>
39  	 * 
40  	 * This method delivers its results asynchronously.  If the caller
41  	 * does not have
42  	 * <code>javax.tv.service.ReadPermission(pe.getLocator())</code>
43  	 * (where <code>pe</code> is the current program event), this
44  	 * method will result in an <code>SIRequestFailureType</code> of
45  	 * <code>DATA_UNAVAILABLE</code>.
46  	 * 
47  	 * @param requestor - The SIRequestor to be notified when this retrieval operation completes.
48  	 * @return An SIRequest object identifying this asynchronous retrieval request.
49  	 * @see ProgramEvent, ReadPermission
50  	 */
51  	public SIRequest retrieveCurrentProgramEvent( SIRequestor requestor);
52  
53  	/**
54  	 * Retrieves the program event for the specified time.  The
55  	 * specified time will fall between the resulting program event's
56  	 * start time (inclusive) and end time (exclusive).<p>
57  	 * 
58  	 * This method delivers its results asynchronously.  If the caller
59  	 * does not have
60  	 * <code>javax.tv.service.ReadPermission(pe.getLocator())</code>
61  	 * (where <code>pe</code> is the program event at the specified
62  	 * time), this method will result in an
63  	 * <code>SIRequestFailureType</code> of
64  	 * <code>DATA_UNAVAILABLE</code>.
65  	 * 
66  	 * @param time - The time of the program event to be retrieved.
67  	 * @param requestor - The SIRequestor to be notified when this retrieval operation completes.
68  	 * @return An SIRequest object identifying this asynchronous retrieval request.
69  	 * @throws SIException - If time does not represent a future time value.
70  	 * @see ProgramEvent, ReadPermission
71  	 */
72  	public SIRequest retrieveFutureProgramEvent(java.util.Date time, SIRequestor requestor) throws SIException;
73  
74  	/**
75  	 * Retrieves all known program events on this service for the
76  	 * specified time interval.  A program event <code>pe</code> is
77  	 * retrieved by this method if the time interval from
78  	 * <code>pe.getStartTime()</code> (inclusive) to
79  	 * <code>pe.getEndTime()</code> (exclusive) intersects the time
80  	 * interval from <code>begin</code> (inclusive) to <code>end</code>
81  	 * (exclusive) specified by the input parameters.<p>
82  	 * 
83  	 * This method returns data asynchronously.  Only program events
84  	 * <code>pe</code> for which the caller has
85  	 * <code>javax.tv.service.ReadPermission(pe.getLocator())</code>
86  	 * will be retrieved.  If no program events meet this criteria,
87  	 * this method will result in an <code>SIRequestFailureType</code> of
88  	 * <code>DATA_UNAVAILABLE</code>.
89  	 * 
90  	 * @param begin - Time identifying the beginning of the interval.
91  	 * @param end - Time identifying the end of the interval.
92  	 * @param requestor - The SIRequestor to be notified when this retrieval operation completes.
93  	 * @return An SIRequest object identifying this asynchronous retrieval request.
94  	 * @throws SIException - If end represents a time value before begin, or if end does not represent a future time value.
95  	 * @see ProgramEvent, ReadPermission
96  	 */
97  	public SIRequest retrieveFutureProgramEvents(java.util.Date begin, java.util.Date end, SIRequestor requestor) throws SIException;
98  
99  	/**
100 	 * Retrieves a event which follows the specified event.<p>
101 	 * 
102 	 * This method delivers its results asynchronously.  If the caller
103 	 * does not have
104 	 * <code>javax.tv.service.ReadPermission(pe.getLocator())</code>
105 	 * (where <code>pe</code> is the next program event), this
106 	 * method will result in an <code>SIRequestFailureType</code> of
107 	 * <code>DATA_UNAVAILABLE</code>.
108 	 * 
109 	 * @param event - A reference event.
110 	 * @param requestor - The SIRequestor to be notified when this retrieval operation completes.
111 	 * @return An SIRequest object identifying this asynchronous retrieval request.
112 	 * @throws SIException - If event does not belong to this ProgramSchedule.
113 	 * @see ProgramEvent, ReadPermission
114 	 */
115 	public SIRequest retrieveNextProgramEvent( ProgramEvent event, SIRequestor requestor) throws SIException;
116 
117 	/**
118 	 * Retrieves a program event matching the locator. Note that
119 	 * the event must be part of this schedule.<p>
120 	 * 
121 	 * This method returns data asynchronously.
122 	 * 
123 	 * @param locator - Locator referencing the ProgramEvent of interest.
124 	 * @param requestor - The SIRequestor to be notified when this retrieval operation completes.
125 	 * @return An SIRequest object identifying this asynchronous retrieval request.
126 	 * @throws InvalidLocatorException - If locator does not reference a valid ProgramEvent in this ProgramSchedule.
127 	 * @throws java.lang.SecurityException - If the caller does not have javax.tv.service.ReadPermission(locator).
128 	 * @see ProgramEvent, ReadPermission
129 	 */
130 	public SIRequest retrieveProgramEvent( Locator locator, SIRequestor requestor) throws InvalidLocatorException, java.lang.SecurityException;
131 
132 	/**
133 	 * Registers a <code>ProgramScheduleListener</code> to be notified of
134 	 * changes to program events on this <code>ProgramSchedule</code>.
135 	 * Subsequent changes will be indicated through instances of
136 	 * <code>ProgramScheduleEvent</code>, with this
137 	 * <code>ProgramSchedule</code> as the event source and an
138 	 * <code>SIChangeType</code> of <code>ADD</code>,
139 	 * <code>REMOVE</code>, <code>MODIFY</code>, or
140 	 * <code>CURRENT_PROGRAM_EVENT</code>.  Only changes to
141 	 * <code>ProgramEvent</code> instances <code>p</code> for which the
142 	 * caller has
143 	 * <code>javax.tv.service.ReadPermission(p.getLocator())</code> will
144 	 * be reported.<p>
145 	 * 
146 	 * This method is only a request for notification.  No guarantee is
147 	 * provided that the SI database will detect all, or even any,
148 	 * changes to the <code>ProgramSchedule</code>, or whether such changes
149 	 * will be detected in a timely fashion.<p>
150 	 * 
151 	 * If the specified <code>ProgramScheduleListener</code> is already
152 	 * registered, no action is performed.
153 	 * 
154 	 * @param listener - A ProgramScheduleListener to be notified of changes to program events on this ProgramSchedule.
155 	 * @see ProgramEvent, ProgramScheduleEvent, ProgramScheduleChangeType, ReadPermission
156 	 */
157 	public void addListener( ProgramScheduleListener listener);
158 
159 	/**
160 	 * Unregisters a <code>ProgramScheduleListener</code>.  If the
161 	 * specified <code>ProgramScheduleListener</code> is not registered, no
162 	 * action is performed.
163 	 * 
164 	 * @param listener - A previously registered listener.
165 	 */
166 	public void removeListener( ProgramScheduleListener listener);
167 
168 	/**
169 	 * Reports the transport-dependent locator referencing the service to
170 	 * which this <code>ProgramSchedule</code> belongs.  Note that
171 	 * applications may use this method to establish the identity of
172 	 * a <code>ProgramSchedule</code> after it has changed.
173 	 * 
174 	 * @return The transport-dependent locator referencing the service to which this ProgramSchedule belongs.
175 	 * @see ProgramScheduleEvent.getProgramSchedule()
176 	 */
177 	public Locator getServiceLocator();
178 
179 }
