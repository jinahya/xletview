14  package xjavax.tv.service.selection;
15  
16  /**
17   * <code>NormalContentEvent</code> is generated to indicate that the
18   * normal content of a service is being presented. This event will be
19   * generated in two situations: <ul>
20   * 
21   * <li>
22   * At the end of a successful service selection operation, this
23   * event will be generated as long as all of the service components
24   * being presented are part of the normal content of the service. The
25   * generation of this event in this situation signals a change in
26   * state of the service context from the <em>presentation pending</em>
27   * state to the <em>presenting</em> state. If even one of the service
28   * components being presented is alternative content (as defined in
29   * <code>AlternativeContentEvent</code>) then that event will be
30   * generated instead.
31   * </li>
32   * 
33   * <li>
34   * During the presentation of a service, this event will be generated
35   * if alternative content was being presented and all of that
36   * alternative content is replaced by content which is a normal part
37   * of the service.  An example of this is when an
38   * end user completes a pay per view dialog the video
39   * advertising a service is replaced by the service itself.
40   * </li>
41   * </ul>
42   * <HR>
43   * 
44   * @author Martin Sveden
45   * @statuscode 4
46   */
47  public class NormalContentEvent extends PresentationChangedEvent{
48  	/**
49  	 * Constructs the event.
50  	 * 
51  	 * @param source - The ServiceContext that generated the event.
52  	 */
53  	public NormalContentEvent( ServiceContext source){
54  		super(source);
55  	}
56  
57  }
