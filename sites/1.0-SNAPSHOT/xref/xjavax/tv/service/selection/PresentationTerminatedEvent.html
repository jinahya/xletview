14  package xjavax.tv.service.selection;
15  
16  /**
17   * <code>PresentationTerminatedEvent</code> is generated when the
18   * presentation of a service terminates.  This includes both normal
19   * termination (e.g., due to an application calling the <code>stop()</code>
20   * method) and abnormal termination (e.g., due to some change in the
21   * environment). Examples of abnormal termination include:
22   * 
23   * <ul>
24   * <li>a tuning operation making the service unavailable </li>
25   * 
26   * <li>removal of fundamental resources required to present the
27   * service</li>
28   * 
29   * <li>withdrawal of CA authorization</li>
30   * </ul>
31   * 
32   * <code>PresentationTerminatedEvent</code> is also generated
33   * following a <code>SelectionFailedEvent</code> either if the service
34   * context was not previously in the <em>presenting</em> state or if
35   * recovery of what was being presented previously is not possible.
36   * <code>PresentationTerminatedEvent</code> is only generated when no
37   * components of the requested service can be presented.  <p>
38   * 
39   * Once this event has been generated, a <code>ServiceContext</code>
40   * will be in the <em>not presenting</em> state until a call to a
41   * <code>select()</code> method succeeds.  When this event is
42   * generated, all resources used for the presentation have been
43   * released, and <code>ServiceContentHandler</code> instances
44   * previously associated with the <code>ServiceContext</code> will
45   * have ceased presentation of their content.
46   * <HR>
47   * 
48   * @author Martin Sveden
49   * @statuscode 4
50   */
51  public class PresentationTerminatedEvent extends ServiceContextEvent{
52  	/**
53  	 * Reason code : The service vanished from the network.</DL>
54  	 * 
55  	 */
56  	public static final int SERVICE_VANISHED = 1;
57  
58  	/**
59  	 * Reason code : Tuning made the service unavailable.</DL>
60  	 * 
61  	 */
62  	public static final int TUNED_AWAY = 2;
63  
64  	/**
65  	 * Reason code : Resources needed to present the service have been removed.</DL>
66  	 * 
67  	 */
68  	public static final int RESOURCES_REMOVED = 3;
69  
70  	/**
71  	 * Reason code : Access to the service or some component of it has been
72  	 * withdrawn by the system. An example of this is the end of a free
73  	 * preview period for IPPV content.</DL>
74  	 * 
75  	 */
76  	public static final int ACCESS_WITHDRAWN = 4;
77  
78  	/**
79  	 * Reason code : The user requested that the presentation be stopped.</DL>
80  	 * 
81  	 * 
82  	 */
83  	public static final int USER_STOP = 5;
84  
85  	private int reason;
86  
87  	/**
88  	 * Constructs the event with a reason code.
89  	 * 
90  	 * @param source - The ServiceContext that generated the event.
91  	 * @param reason - The reason for which the presentation was terminated.
92  	 */
93  	public PresentationTerminatedEvent( ServiceContext source, int reason)	{
94  		super(source);
95  		this.reason = reason;
96  	}
97  
98  	/**
99  	 * Reports the reason for which the presentation was terminated.
100 	 * 
101 	 * @return A reason code for why the presentation was terminated.
102 	 */
103 	public int getReason()	{
104 		return this.reason;
105 	}
106 
107 }
