14  package xjavax.tv.service;
15  
16  /**
17   * The <code>RatingDimension</code> interface represents an
18   * individual content rating scheme against which program events are
19   * rated.  Each rating region may support multiple rating
20   * dimensions. One dimension in the U.S. rating region, for example,
21   * is used to describe the MPAA list. The dimension name for such a
22   * case may be defined as "MPAA".  Another example of a rating
23   * dimension may be an age-based DVB rating.
24   * <A HREF="../../../javax/tv/service/guide/ContentRatingAdvisory.html"><CODE>ContentRatingAdvisory</CODE></A></DL>
25   * <HR>
26   * 
27   * 
28   */
29  public interface RatingDimension
30  {
31  	/**
32  	 * Returns a string which represents the dimension name being described by
33  	 * this object. One dimension in the U.S. rating region, for example, is
34  	 * used to describe the MPAA list. The dimension name for such a case may
35  	 * be defined as "MPAA".
36  	 * 
37  	 * @return A string representing the name of this rating dimension.
38  	 */
39  	public java.lang.String getDimensionName();
40  
41  	/**
42  	 * Returns the number of levels defined for this dimension.
43  	 * 
44  	 * @return The number of levels in this dimension.
45  	 */
46  	public short getNumberOfLevels();
47  
48  	/**
49  	 * Returns a pair of strings describing the specified rating level for
50  	 * this dimension.
51  	 * 
52  	 * @param ratingLevel - The rating level for which to retrieve the textual description.
53  	 * @return A pair of strings representing the names for the specified rating level. The first string represents the abbreviated name for the rating level. The second string represents the full name for the rating level.
54  	 * @throws SIException - If ratingLevel is not valid for this RatingDimension.
55  	 * @see ContentRatingAdvisory.getRatingLevel(java.lang.String)
56  	 */
57  	public java.lang.String[] getRatingLevelDescription(short ratingLevel) throws SIException;
58  
59  }
