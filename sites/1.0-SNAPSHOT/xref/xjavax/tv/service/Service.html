14  package xjavax.tv.service;
15  
16  import xjavax.tv.locator.Locator;
17  
18  /**
19   * The <code>Service</code> interface represents an abstract view on
20   * what is generally referred to as a television "service" or
21   * "channel". It may represent an MPEG-2 program, DVB service, an ATSC
22   * virtual channel, SCTE virtual channel, etc. It represents the basic
23   * information associated with a service, such as its name or number,
24   * which is guaranteed to be available on the receiver. <P>
25   * 
26   * Internal to the receiver, each service is uniquely identified by
27   * information that may include system type, network ID, transport stream
28   * ID, service number, service number, or other information. This
29   * identification is encapsulated by the Locator object. <P>
30   * 
31   * Note that a <code>Service</code> object may represent multiple
32   * instances of the same content delivered over different media
33   * (e.g., the same service may be delivered over a terrestrial and
34   * cable network). A <code>ServiceDetails</code> object represents a
35   * specific instance of such content which is bound to a specific
36   * delivery mechanism. <P>
37   * 
38   * The information available through this object, i.e., the service name,
39   * service number, etc., represents information that is stored in
40   * the receiver and is not necessarily the same as what is broadcast in any
41   * broadcast service information protocol. For example, a receiver
42   * implementation may let the end user edit this information according to
43   * the user's preferences. <P>
44   * 
45   * A <code>Service</code> object may optionally implement an interface
46   * that supports service numbers.  Each <code>Service</code> object
47   * must provide either a service name (via the <code>getName</code> method)
48   * or a service number (via the <code>ServiceNumber</code> interface).
49   * <A HREF="../../../javax/tv/service/ServiceNumber.html"><CODE>ServiceNumber</CODE></A>,
50   * <A HREF="../../../javax/tv/service/navigation/ServiceDetails.html"><CODE>ServiceDetails</CODE></A>,
51   * <a href="../../../overview-summary.html#guidelines-opinterfaces">Optionally implemented interfaces</a></DL>
52   * <HR>
53   * 
54   * 
55   */
56  public interface Service
57  {
58  	/**
59  	 * This method retrieves additional information about the
60  	 * <code>Service</code>. This information is retrieved from the
61  	 * broadcast service information. <P>
62  	 * 
63  	 * Note that if the content represented by this <code>Service</code>
64  	 * is delivered on multiple transport-dependent streams there may be
65  	 * multiple <code>ServiceDetails</code> for it. This method
66  	 * retrieves one of them based on availability or user
67  	 * preferences. If access to all possible
68  	 * <code>ServiceDetails</code> is required, the service
69  	 * <code>Locator</code> can be transformed to transport-dependent
70  	 * <code>Locator</code> instances and <code>ServiceDetails</code>
71  	 * can be retrieved for each. <P>
72  	 * 
73  	 * This method returns data asynchronously.
74  	 * 
75  	 * @param requestor - The SIRequestor to be notified when this retrieval operation completes.
76  	 * @return An SIRequest object identifying this asynchronous retrieval request.
77  	 * @see Locator, ServiceDetails
78  	 */
79  	public SIRequest retrieveDetails( SIRequestor requestor);
80  
81  	/**
82  	 * Returns a short service name or acronym.  For example, in ATSC
83  	 * systems the service name is provided by the the PSIP VCT; in DVB
84  	 * systems, this information is provided by the DVB Service
85  	 * Descriptor or the Multilingual Service Name Descriptor.  The
86  	 * service name may also be user-defined.
87  	 * 
88  	 * @return A string representing this service's short name.  If the short name is unavailable, the string representation of the service number is returned.
89  	 */
90  	public java.lang.String getName();
91  
92  	/**
93  	 * This method indicates whether the service represented by this
94  	 * <code>Service</code> object is available on multiple
95  	 * transports, (e.g., the same content delivered over terrestrial and
96  	 * cable network).
97  	 * 
98  	 * @return true if multiple transports carry the same content identified by this Service object; false if there is only one instance of this service.
99  	 */
100 	public boolean hasMultipleInstances();
101 
102 	/**
103 	 * Returns the type of this service, (for example, "digital
104 	 * television", "digital radio", "NVOD", etc.) These values can be
105 	 * mapped to the ATSC service type in the VCT table and the DVB
106 	 * service type in the service descriptor.
107 	 * 
108 	 * @return Service type of this Service.
109 	 */
110 	public ServiceType getServiceType();
111 
112 	/**
113 	 * Reports the <code>Locator</code> of this <code>Service</code>.
114 	 * Note that if the resulting locator is transport-dependent, it
115 	 * will also correspond to a <code>ServiceDetails</code> object.
116 	 * 
117 	 * @return A locator referencing this Service.
118 	 * @see ServiceDetails
119 	 */
120 	public Locator getLocator();
121 
122 	/**
123 	 * Tests two <code>Service</code> objects for equality.  Returns
124 	 * <code>true</code> if and only if:
125 	 * <ul>
126 	 * <li><code>obj</code>'s class is the
127 	 * same as the class of this <code>Service</code>, and<p>
128 	 * <li><code>obj</code>'s <code>Locator</code> is equal to
129 	 * the <code>Locator</code> of this <code>Service</code>
130 	 * (as reported by
131 	 * <code>Service.getLocator()</code>, and<p>
132 	 * <li><code>obj</code> and this object encapsulate identical data.
133 	 * </ul>
134 	 * 
135 	 * @param obj - The object against which to test for equality.
136 	 * @return true if the two Service objects are equal; false otherwise.
137 	 * @see equals in class java.lang.Object
138 	 */
139 	public boolean equals(java.lang.Object obj);
140 
141 	/**
142 	 * Reports the hash code value of this <code>Service</code>.  Two
143 	 * <code>Service</code> objects that are equal will have identical
144 	 * hash codes.
145 	 * 
146 	 * @return The hash code value of this Service.
147 	 * @see hashCode in class java.lang.Object
148 	 */
149 	public int hashCode();
150 
151 }
