14  package xjavax.tv.service;
15  
16  import xjavax.tv.locator.Locator;
17  
18  /**
19   * The base interface of elements provided by the SI database.
20   * 
21   * The base interface of elements provided by the SI database.
22   * <code>SIElement</code> objects represent immutable <em>copies</em>
23   * of the service information data contained in the SI database.  If
24   * the information represented by an <code>SIElement</code> <em>E</em>
25   * changes in the database, <em>E</em> will not be changed.  The value
26   * of the <code>SIElement</code>'s locator (obtained by the
27   * <code>getLocator()</code> method) will remain unchanged in this
28   * case; the locator may be used to retrieve a copy of the SI element
29   * with the new data.  Two <code>SIElement</code> objects retrieved
30   * from the SI database using the same input <code>Locator</code> at
31   * different times will report <code>Locator</code> objects that are
32   * equal according to <code>Locator.equal()</code>.  However, the
33   * <code>SIElement</code> objects themselves will not be
34   * <code>equal()</code> if the corresponding data changed in the SI
35   * database between the times of their respective retrievals.
36   * <A HREF="../../../javax/tv/service/SIManager.html#retrieveSIElement(javax.tv.locator.Locator, javax.tv.service.SIRequestor)"><CODE>SIManager.retrieveSIElement(javax.tv.locator.Locator, javax.tv.service.SIRequestor)</CODE></A></DL>
37   * <HR>
38   * 
39   * 
40   */
41  public interface SIElement extends SIRetrievable
42  {
43  	/**
44  	 * Reports the <code>Locator</code> of this <code>SIElement</code>.
45  	 * 
46  	 * @return Locator The locator referencing this SIElement
47  	 */
48  	public Locator getLocator();
49  
50  	/**
51  	 * Tests two <code>SIElement</code> objects for equality.  Returns
52  	 * <code>true</code> if and only if:
53  	 * <ul>
54  	 * <li><code>obj</code>'s class is the
55  	 * same as the class of this <code>SIElement</code>, and<p>
56  	 * <li><code>obj</code>'s <code>Locator</code> is equal to
57  	 * the <code>Locator</code> of this object (as reported by
58  	 * <code>SIElement.getLocator()</code>, and<p>
59  	 * <li><code>obj</code> and this object encapsulate identical data.
60  	 * </ul>
61  	 * 
62  	 * @param obj - The object against which to test for equality.
63  	 * @return true if the two SIElement objects are equal; false otherwise.
64  	 * @see equals in class java.lang.Object
65  	 */
66  	public boolean equals(java.lang.Object obj);
67  
68  	/**
69  	 * Reports the hash code value of this <code>SIElement</code>.  Two
70  	 * <code>SIElement</code> objects that are equal will have identical
71  	 * hash codes.
72  	 * 
73  	 * @return The hash code value of this SIElement.
74  	 * @see hashCode in class java.lang.Object
75  	 */
76  	public int hashCode();
77  
78  	/**
79  	 * Reports the SI format in which this <code>SIElement</code> was
80  	 * delivered.
81  	 * 
82  	 * @return The SI format in which this SI element was delivered.
83  	 */
84  	public ServiceInformationType getServiceInformationType();
85  
86  }
