14  package xjavax.tv.service.navigation;
15  
16  import xjavax.tv.service.SIElement;
17  import xjavax.tv.service.SIRequest;
18  import xjavax.tv.service.SIRequestor;
19  import xjavax.tv.service.Service;
20  import xjavax.tv.service.ServiceType;
21  import xjavax.tv.service.guide.ProgramSchedule;
22  
23  /**
24   * This interface provides access to service meta-data.
25   * 
26   * This interface provides access to service meta-data. It provides more
27   * information about a <code>Service</code> object and represents a
28   * specific instance of a service bound to a transport stream. <p>
29   * 
30   * A <code>ServiceDetails</code> object may optionally implement the
31   * <code>ServiceNumber</code> interface to report service numbers as
32   * assigned by the broadcaster of the service.<p>
33   * 
34   * A <code>ServiceDetails</code> object may optionally implement the
35   * <code>ServiceProviderInformation</code> interface to report information
36   * concerning the service provider.
37   * <A HREF="../../../../javax/tv/service/ServiceNumber.html"><CODE>ServiceNumber</CODE></A>,
38   * <A HREF="../../../../javax/tv/service/navigation/ServiceProviderInformation.html"><CODE>ServiceProviderInformation</CODE></A>,
39   * <a href="../../../../overview-summary.html#guidelines-opinterfaces">Optionally implemented interfaces</a></DL>
40   * <HR>
41   * 
42   * 
43   */
44  public interface ServiceDetails extends SIElement, CAIdentification
45  {
46  	/**
47  	 * Retrieves a textual description of this service if available.
48  	 * This method delivers its results asynchronously.
49  	 * 
50  	 * @param requestor - The SIRequestor to be notified when this retrieval operation completes.
51  	 * @return An SIRequest object identifying this asynchronous retrieval request.
52  	 * @see ServiceDescription
53  	 */
54  	public SIRequest retrieveServiceDescription( SIRequestor requestor);
55  
56  	/**
57  	 * Returns the type of this service, for example, "digital
58  	 * television", "digital radio", "NVOD", etc. These values can be
59  	 * mapped to the ATSC service type in the VCT table and the DVB
60  	 * service type in the Service Descriptor.
61  	 * 
62  	 * @return Service type of this service.
63  	 */
64  	public ServiceType getServiceType();
65  
66  	/**
67  	 * Retrieves an array of elementary components which are part of
68  	 * this service.  The array will only contain
69  	 * <code>ServiceComponent</code> instances <code>c</code> for which
70  	 * the caller has
71  	 * <code>javax.tv.service.ReadPermission(c.getLocator())</code>.  If
72  	 * no <code>ServiceComponent</code> instances meet this criteria,
73  	 * this method will result in an <code>SIRequestFailureType</code> of
74  	 * <code>DATA_UNAVAILABLE</code>.
75  	 * 
76  	 * This method delivers its results asynchronously.
77  	 * 
78  	 * @param requestor - The SIRequestor to be notified when this retrieval operation completes.
79  	 * @return An SIRequest object identifying this asynchronous retrieval request.
80  	 * @see ServiceComponent, ReadPermission
81  	 */
82  	public SIRequest retrieveComponents( SIRequestor requestor);
83  
84  	/**
85  	 * Returns a schedule of program events associated with this service.
86  	 * 
87  	 * @return The program schedule for this service, or null if no schedule is available.
88  	 */
89  	public ProgramSchedule getProgramSchedule();
90  
91  	/**
92  	 * Called to obtain a full service name. For example, this
93  	 * information may be delivered in the ATSC Extended Channel Name
94  	 * Descriptor, the DVB Service Descriptor or the DVB Multilingual
95  	 * Service Name Descriptor.
96  	 * 
97  	 * @return A string representing the full service name, or an empty string if the name is not available.
98  	 */
99  	public java.lang.String getLongName();
100 
101 	/**
102 	 * Returns the <code>Service</code> this <code>ServiceDetails</code>
103 	 * object is associated with.
104 	 * 
105 	 * @return The Service to which this ServiceDetails belongs.
106 	 */
107 	public Service getService();
108 
109 	/**
110 	 * Registers a <code>ServiceComponentChangeListener</code> to be
111 	 * notified of changes to a <code>ServiceComponent</code> that is
112 	 * part of this <code>ServiceDetails</code>. Subsequent notification
113 	 * is made via <code>ServiceComponentChangeEvent</code> with this
114 	 * <code>ServiceDetails</code> instance as the event source and an
115 	 * <code>SIChangeType</code> of <code>ADD</code>,
116 	 * <code>REMOVE</code> or <code>MODIFY</code>. Only changes to
117 	 * <code>ServiceComponent</code> instances <code>c</code> for which
118 	 * the caller has
119 	 * <code>javax.tv.service.ReadPermission(c.getLocator())</code> will
120 	 * be reported.<p>
121 	 * 
122 	 * This method is only a request for notification.  No guarantee is
123 	 * provided that the SI database will detect all, or even any, SI
124 	 * changes or whether such changes will be detected in a timely
125 	 * fashion.<p>
126 	 * 
127 	 * If the specified <code>ServiceComponentChangeListener</code> is
128 	 * already registered, no action is performed.
129 	 * 
130 	 * @param listener - A ServiceComponentChangeListener to be notified about changes related to a ServiceComponent in this ServiceDetails.
131 	 * @see ServiceComponentChangeEvent, ReadPermission
132 	 */
133 	public void addServiceComponentChangeListener( ServiceComponentChangeListener listener);
134 
135 	/**
136 	 * Called to unregister an
137 	 * <code>ServiceComponentChangeListener</code>.  If the specified
138 	 * <code>ServiceComponentChangeListener</code> is not registered, no
139 	 * action is performed.
140 	 * 
141 	 * @param listener - A previously registered listener.
142 	 */
143 	public void removeServiceComponentChangeListener( ServiceComponentChangeListener listener);
144 
145 	/**
146 	 * Reports the type of mechanism by which this service was
147 	 * delivered.
148 	 * 
149 	 * @return The delivery system type of this service.
150 	 */
151 	public DeliverySystemType getDeliverySystemType();
152 
153 }
