14  package xjavax.tv.service.navigation;
15  
16  /**
17   * This interface associates information related to
18   * the conditional access (CA) subsystem with certain SI objects.
19   * <A HREF="../../../../javax/tv/service/guide/ProgramEvent.html"><CODE>ProgramEvent</CODE></A>,
20   * <A HREF="../../../../javax/tv/service/transport/Bouquet.html"><CODE>Bouquet</CODE></A></DL>
21   * <HR>
22   * 
23   * 
24   */
25  public interface CAIdentification
26  {
27  	/**
28  	 * Returns an array of CA System IDs associated with this object. This
29  	 * information may be obtained from the CAT MPEG message or a system
30  	 * specific conditional access descriptor (such as defined by Simulcrypt
31  	 * or ATSC).
32  	 * 
33  	 * @return An array of CA System IDs. An empty array is returned when no CA System IDs are available.
34  	 */
35  	public int[] getCASystemIDs();
36  
37  	/**
38  	 * Provides information concerning conditional access of this object.
39  	 * 
40  	 * @return true if this Service is not protected by a conditional access; false if one or more components is protected by conditional access.
41  	 */
42  	public boolean isFree();
43  
44  }
