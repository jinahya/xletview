14  package xjavax.tv.service.navigation;
15  
16  import xjavax.tv.locator.InvalidLocatorException;
17  import xjavax.tv.locator.Locator;
18  import xjavax.tv.service.Service;
19  
20  /**
21   * <code>ServiceList</code> represents an ordered list of
22   * <code>Service</code> objects based on a specific grouping rule
23   * defined by a <code>ServiceFilter</code>.  The objects in a
24   * <code>ServiceList</code> are numbered from 0 to <code>size()
25   * -1</code>.
26   * 
27   * A <code>ServiceList</code> is <i>immutable</i>.  In other words,
28   * once a <code>ServiceList</code> instance is created, the elements
29   * in the list and their order will never change.  All classes that
30   * implement the <code>ServiceList</code> interface are required to
31   * maintain this property.
32   * <A HREF="../../../../javax/tv/service/navigation/ServiceFilter.html"><CODE>ServiceFilter</CODE></A>,
33   * <A HREF="../../../../javax/tv/service/navigation/ServiceList.html#size()"><CODE>size()</CODE></A></DL>
34   * <HR>
35   * 
36   * 
37   */
38  public interface ServiceList
39  {
40  	/**
41  	 * Generates a new <code>ServiceList</code> containing the
42  	 * same elements as the current list, sorted in ascending
43  	 * order by service name.
44  	 * 
45  	 * @return A ServiceList sorted by service name.
46  	 * @see Service.getName()
47  	 */
48  	public ServiceList sortByName();
49  
50  	/**
51  	 * Generates a new <code>ServiceList</code> containing the
52  	 * same elements as the current list, sorted in ascending
53  	 * order by service number.
54  	 * 
55  	 * @return A ServiceList sorted by service number.
56  	 * @throws SortNotAvailableException - If any of the Service objects in this ServiceList do not implement the ServiceNumber interface.
57  	 * @see ServiceNumber
58  	 */
59  	public ServiceList sortByNumber() throws SortNotAvailableException;
60  
61  	/**
62  	 * Reports the <code>Service</code> corresponding to the specified
63  	 * locator if it is a member of this list.
64  	 * 
65  	 * @param locator - Specifies the Service to be searched for.
66  	 * @return The Service corresponding to locator, or null if the Service is not a member of this list.
67  	 * @throws InvalidLocatorException - If locator does not reference a valid Service.
68  	 */
69  	public Service findService( Locator locator) throws InvalidLocatorException;
70  
71  	/**
72  	 * Creates a new <code>ServiceList</code> object that is a subset of
73  	 * this list, based on the conditions specified by a
74  	 * <code>ServiceFilter</code> object. This method may be used to
75  	 * generate increasingly specialized lists of <code>Service</code>
76  	 * objects based on multiple filtering criteria.  If the filter is
77  	 * <code>null</code>, the resulting <code>ServiceList</code> will be
78  	 * a duplicate of this list.  <p>
79  	 * 
80  	 * Note that the <code>accept</code> method of the given
81  	 * <code>ServiceFilter</code> will be invoked for each
82  	 * <code>Service</code> to be filtered using the same application
83  	 * thread that invokes this method.
84  	 * 
85  	 * @param filter - A filter constraining the requested service list, or null.
86  	 * @return A ServiceList object created based on the specified filtering rules.
87  	 * @see ServiceFilter.accept(javax.tv.service.Service)
88  	 */
89  	public ServiceList filterServices( ServiceFilter filter);
90  
91  	/**
92  	 * Generates an iterator on the <code>Service</code> elements
93  	 * in this list.
94  	 * 
95  	 * @return A ServiceIterator on the Service s in this list.
96  	 */
97  	public ServiceIterator createServiceIterator();
98  
99  	/**
100 	 * Tests if the indicated <code>Service</code> object is contained
101 	 * in the list.
102 	 * 
103 	 * @param service - The Service object for which to search.
104 	 * @return true if the specified Service is member of the list; false otherwise.
105 	 */
106 	public boolean contains( Service service);
107 
108 	/**
109 	 * Reports the position of the first occurrence of the
110 	 * indicated <code>Service</code> object in the list.
111 	 * 
112 	 * @param service - The Service object for which to search.
113 	 * @return The index of the first occurrence of the service, or -1 if service is not contained in the list.
114 	 */
115 	public int indexOf( Service service);
116 
117 	/**
118 	 * Reports the number of <code>Service</code> objects in the list.
119 	 * 
120 	 * @return The number of Service objects in the list.
121 	 */
122 	public int size();
123 
124 	/**
125 	 * Reports the <code>Service</code> at the specified index position.
126 	 * 
127 	 * @param index - A position in the ServiceList.
128 	 * @return The Service at the specified index.
129 	 * @throws java.lang.IndexOutOfBoundsException - If index index > size()-1.
130 	 */
131 	public Service getService(int index);
132 
133 	/**
134 	 * Compares the specified object with this <code>ServiceList</code>
135 	 * for equality. Returns <code>true</code> if and only if the
136 	 * specified object is also a <code>ServiceList</code>, both lists
137 	 * have the same size, and all corresponding pairs of elements in
138 	 * the two lists are equal.  (Two elements e1 and e2 are equal if
139 	 * (e1==null ? e2==null : e1.equals(e2)).) In other words, two lists
140 	 * are defined to be equal if they contain the same elements in the
141 	 * same order.
142 	 * 
143 	 * @param o - The object to be compared for equality with this list.
144 	 * @return true if the specified object is equal to this list; false otherwise.
145 	 * @see equals in class java.lang.Object
146 	 */
147 	public boolean equals(java.lang.Object o);
148 
149 	/**
150 	 * Provides the hash code value for this <code>ServiceList</code>.
151 	 * Two <code>ServiceList</code> objects that are equal will have
152 	 * the same hash code.
153 	 * 
154 	 * @return The hash code value of this ServiceList.
155 	 * @see hashCode in class java.lang.Object
156 	 */
157 	public int hashCode();
158 
159 }
