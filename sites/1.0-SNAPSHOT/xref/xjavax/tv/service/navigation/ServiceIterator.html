14  package xjavax.tv.service.navigation;
15  
16  import xjavax.tv.service.Service;
17  
18  /**
19   * <code>ServiceIterator</code> permits iteration over an ordered
20   * list of <code>Service</code> objects.  Applications may use the
21   * <code>ServiceIterator</code> interface to browse a
22   * <code>ServiceList</code> forward or backward.<p>
23   * 
24   * Upon initial usage, <code>hasPrevious()</code> will return
25   * <code>false</code> and <code>nextService()</code> will return the
26   * first <code>Service</code> in the list, if present.
27   * <HR>
28   * 
29   * 
30   */
31  public interface ServiceIterator
32  {
33  	/**
34  	 * Resets the iterator to the beginning of the list, such that
35  	 * <code>hasPrevious()</code> returns <code>false</code> and
36  	 * <code>nextService()</code> returns the first <code>Service</code>
37  	 * in the list (if the list is not empty).</DL>
38  	 * 
39  	 */
40  	public void toBeginning();
41  
42  	/**
43  	 * Sets the iterator to the end of the list, such that
44  	 * <code>hasNext()</code> returns <code>false</code> and
45  	 * <code>previousService()</code> returns the last <code>Service</code>
46  	 * in the list (if the list is not empty).</DL>
47  	 * 
48  	 */
49  	public void toEnd();
50  
51  	/**
52  	 * Reports the next <code>Service</code> object in the list.  This
53  	 * method may be called repeatedly to iterate through the list.
54  	 * 
55  	 * @return The Service object at the next position in the list.
56  	 * @throws java.util.NoSuchElementException - If the iteration has no next Service.
57  	 */
58  	public Service nextService();
59  
60  	/**
61  	 * Reports the previous <code>Service</code> object in the list.
62  	 * This method may be called repeatedly to iterate through the list
63  	 * in reverse order.
64  	 * 
65  	 * @return The Service object at the previous position in the list.
66  	 * @throws java.util.NoSuchElementException - If the iteration has no previous Service.
67  	 */
68  	public Service previousService();
69  
70  	/**
71  	 * Tests if there is a <code>Service</code> in the next position in
72  	 * the list.
73  	 * 
74  	 * @return true if there is a Service in the next position in the list; false otherwise.
75  	 */
76  	public boolean hasNext();
77  
78  	/**
79  	 * Tests if there is a <code>Service</code> in the previous
80  	 * position in the list.
81  	 * 
82  	 * @return true if there is a Service in the previous position in the list; false otherwise.
83  	 */
84  	public boolean hasPrevious();
85  
86  }
