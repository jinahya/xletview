14  package xjavax.tv.service;
15  
16  /**
17   * This interface is implemented by application classes to receive the
18   * results of asynchronous SI retrieval requests. The
19   * <code>SIRequestor</code> registers itself at the time of the
20   * asynchronous call for a single request and is automatically
21   * unregistered when the request is completed.  Applications can
22   * disambiguate retrieval operations by registering a unique
23   * <code>SIRequestor</code> for each retrieval request.<p>
24   * 
25   * The asynchronous SI retrieval mechanisms invoke the methods of this
26   * interface using system threads that are guaranteed to not hold
27   * locks on application objects.
28   * <HR>
29   * 
30   * 
31   */
32  public interface SIRequestor
33  {
34  	/**
35  	 * Notifies the <code>SIRequestor</code> of successful asynchronous
36  	 * SI retrieval.
37  	 * 
38  	 * @param result - The previously requested data.
39  	 */
40  	public void notifySuccess( SIRetrievable[] result);
41  
42  	/**
43  	 * Notifies the <code>SIRequestor</code> of unsuccessful asynchronous
44  	 * SI retrieval.
45  	 * 
46  	 * @param reason - The reason why the asynchronous request failed.
47  	 */
48  	public void notifyFailure( SIRequestFailureType reason);
49  
50  }
