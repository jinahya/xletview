66      public TvWindow() {
67          super();
68          
69          createMenu();
70          Container container = getContentPane();
71          container.setLayout(new BorderLayout());
72  
73          int screenWidth;
74          int screenHeight;
75          int x = 0;
76          int y = 0;
77  
78          boolean doCenter = true;
79          
80          try {
81              String strWidth = Settings.getProperty("tv.screenwidth");
82              screenWidth = Integer.parseInt(strWidth);
83              String strHeight = Settings.getProperty("tv.screenheight");
84              screenHeight = Integer.parseInt(strHeight);
85              String strX = Settings.getProperty("tv.x");
86              String strY = Settings.getProperty("tv.y");
87              doCenter = (Settings.getProperty("tv.center").equals("true")) ? true : false;
88              x = Integer.parseInt(strX);
89              y = Integer.parseInt(strY);
90          }
91          catch (Exception e) {
92              log.error(e);
93              screenWidth = Util.parseInt(Settings.getProperty("tv.screenwidth"));
94              screenHeight = Util.parseInt(Settings.getProperty("tv.screenheight"));
95              doCenter = true;
96          }
97          
98          ScreenContainer tv = ScreenContainer.getInstance();
99          tv.setSize(screenWidth, screenHeight);
100         
101         // center the tv screen
102         JPanel centerCont = new JPanel(new CenterLayout());
103         centerCont.setBackground(Color.black);
104         centerCont.add(tv);
105         container.add(centerCont, BorderLayout.CENTER);
106         
107         //container.add(tv, BorderLayout.CENTER);
108 
109         if(Settings.getProperty("remote.show").equalsIgnoreCase("true")){
110             
111         	
112         	try {
113         		//FileInputStream in = new FileInputStream(new File(Util.getURLConnection(ChannelManager.class, "config/remote_control.xml")));
114         		InputStream in = Util.getURLConnection(ChannelManager.class, "config/remote_control.xml").getInputStream();
115         		IXMLReader reader = new StdXMLReader(in);
116         		IXMLParser parser = XMLParserFactory.createDefaultXMLParser();
117         		parser.setReader(reader);
118         		IXMLElement xml = (IXMLElement) parser.parse(); 
119         		RemoteControl remote = RemoteControl.getInstance();
120         		remote.make(xml);
121         		container.add(remote, BorderLayout.EAST);
122         		
123 			} catch (Exception e) {
124 				e.printStackTrace();
125 			}
126         	
127 			
128         }
129 
130         this.pack();
131 
132         addWindowListener(new WindowAdapter() {
133             public void windowClosing(WindowEvent we) {                
134                 // save properties
135                 doClose();
136             }
137             
138             
139         });
140         
141         this.addComponentListener(new ComponentAdapter(){
142             public void componentResized(ComponentEvent e){                
143                 ScreenContainer.getInstance().repaint();
144             }   
145         });
146         
147         addWindowFocusListener(new WindowAdapter() {
148             public void windowLostFocus(WindowEvent we) {                    
149             	EventManager.getInstance().setEventEnabled(false);
150             }
151             public void windowGainedFocus(WindowEvent e) {
152                 EventManager.getInstance().setEventEnabled(true);
153             }
154         });
155 
156         addComponentListener(new ComponentAdapter() {
157             public void componentResized(ComponentEvent e) {
158 
159             }
160         });
161 
162         this.setIconImage(Constants.ICON_TVWINDOW);
163         this.setTitle(Constants.TITLE);
164         this.setResizable(true);  
165               
166 
167         if (doCenter) {
168             Util.center(this);
169         }
170         else{
171             setLocation(x, y);  
172         }
173 
174         // set the first channel
175         ChannelManager.getInstance().setChannel(0);
176         //System.out.println("<end log file>"); 
177         //ConsoleWindow.getInstance();
178 
179         // visibility is set in the startup class
180 
181     }

227     public void actionPerformed(ActionEvent e) {
228         String command = e.getActionCommand();
229         if (e.getSource() instanceof AppMenuItem) {
230             AppMenuItem item = (AppMenuItem) e.getSource();
231             App app = item.getApp();            
232             XletManager.getInstance().setXlet(app.getPath(), app.getXletName());
233         }
234         else if (command.equals("about")) {
235             new AboutWindow(this);
236         }
237         else if (command.equals("exit")) {
238             doClose();
239         }
240         else if (command.equals("console")) {
241             ConsoleWindow.getInstance().show();
242         }
243     }
