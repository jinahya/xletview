61  	public Startup(String[] args) {
62  		
63  		try {
64  			UIManager.setLookAndFeel(new MetouiaLookAndFeel());
65  			UIManager.getLookAndFeelDefaults().put("ClassLoader", this.getClass().getClassLoader());
66  		}
67  		catch (UnsupportedLookAndFeelException exception) {
68  			exception.printStackTrace();
69  		}
70  		
71  		
72  		String xPath = "";
73  		String[] xExtraPaths = null;
74  		String xName = "";
75  
76  		int command = CommandLine.check(args);
77  		if (command == CommandLine.XLET_IS_SET) {
78  			xPath = CommandLine.getXletPath();
79  			xExtraPaths = CommandLine.getXletExtraPaths();
80  			xName = CommandLine.getXletName();
81  		}
82  
83  		// create splash
84  		createSplash();
85  
86  		// do the following on the gui thread
87  		SwingUtilities.invokeLater(new Runnable() {
88  			public void run() {
89  				showSplash();
90  				Startup.this.start = System.currentTimeMillis();
91  			}
92  		});
93  		
94  
95  		this.splash.setMessage("Setting system variables...");
96  
97  		setProperties();
98  
99  		this.splash.setMessage("Initializing windows...");
100 
101 		final TvWindow mainFrame = new TvWindow();
102 
103 		if (xPath.length() > 0 && xName.length() > 0) {
104 			this.splash.setMessage("Setting xlet from command line...");
105 			
106 			URL url = pathString2URL(xPath);
107 			
108 			if (url != null){
109 				Startup.logger.debug("Xlet URL is '"+url.toExternalForm()+"'");
110 			}
111 
112 			if (xExtraPaths != null){
113 				
114 				int numExtraPaths = xExtraPaths.length;
115 				Startup.logger.debug("Xlet has '"+numExtraPaths+"' extra paths...");
116 				
117 				URL[] xExtraPathURLs = new URL[numExtraPaths];
118 				for (int i=0; i<numExtraPaths; i++){
119 					Startup.logger.debug("Processing extra path '"+xExtraPaths[i]+"'.");
120 					xExtraPathURLs[i] = pathString2URL(xExtraPaths[i]);
121 				}
122 				Startup.logger.debug("Processed all extra paths.");
123 				XletManager.getInstance().setXlet(url, xExtraPathURLs, xName);
124 			}
125 			else{ // no Extra paths
126 				Startup.logger.debug("Xlet has NO extra paths.");
127 				XletManager.getInstance().setXlet(url, xName);
128 			}
129 		}
130 
131 		this.splash.setMessage("Loading finished, starting...");
132 
133 		// show prog and hide splash
134 		SwingUtilities.invokeLater(new Runnable() {
135 			public void run() {
136 				Startup.this.end = System.currentTimeMillis();
137 				long diff = Startup.this.minSplashMs - (Startup.this.end - Startup.this.start);
138 				if (diff > 0) {
139 					try {
140 						Thread.sleep(diff);
141 					} catch (InterruptedException e) {
142 						e.printStackTrace();
143 					}
144 				}
145 
146 				mainFrame.show();
147 				mainFrame.toFront();
148 
149 				hideSplash();
150 			}
151 		});
152 
153 	}

155 	/**
156 	 * @param path
157 	 * @return
158 	 */
159 	// TODO: Move somewhere appropriate
160 	public static URL pathString2URL(String path) {
161 		URL url = null;
162 		try{
163 			url = new URL(path);
164 		}
165 		catch (MalformedURLException mue){
166 			Startup.logger.debug("Xlet Path is not an URL, trying to prefix with 'file:'.");
167 			try {
168 				url = new URL("file:"+path);
169 			} catch (MalformedURLException e) {
170 				// TODO Auto-generated catch block
171 				e.printStackTrace();
172 			}
173 		}
174 		return url;
175 	}

190 	private void setProperties() {
191 		try {
192 			Startup.logger.debug("setting properties...");
193 			//String settingsPath = Util.getURLConnection(Startup.class, Constants.PATH_SETTINGS);
194 			//File settingsFile = new File(settingsPath);
195 			//Settings.load(settingsFile);
196 			
197 			InputStream settingsInputStream = Util.getURLConnection(Startup.class, Constants.PATH_SETTINGS).getInputStream();
198 			Settings.load(settingsInputStream);
199 			
200 			
201 			/*
202 			 * I removed the file name normalization here because I found that the Win32 version would not
203 			 * longer be able to find the resources. [Enver <enver.haase@gmx.de> on 07-Jan-2004]
204 			 * See ChannelManager's
205 			 * "media = new Media(Util.getURL(ChannelManager.class, Settings.getProperty("file.defaultbg")));"
206 			 * which would result in "config\defaultbg.jpg" NOT to be found, but "config/defaultbg.jpg" is okay
207 			 * even on a Win32 platform.
208 			 * Funny thing is that under Linux, both variants would be found. Maybe it's an error in the JDK.
209 			 */
210 			Settings.setProperty("path.home", new File("").getAbsolutePath() + File.separator);
211 			/*
212 			Settings.setProperty("path.home", Util.normalizePath(new File("").getAbsolutePath() + File.separator));
213 			Settings.setProperty("file.settings", Util.normalizePath(Settings.getProperty("file.settings")));
214 			Settings.setProperty("file.defaultbg", Util.normalizePath(Settings.getProperty("file.defaultbg")));
215 			*/
216 			
217 			String extraClassPath = Settings.getProperty("extra.classpath");
218 			
219 			ClassLoader classLoader = getClass().getClassLoader();
220 			
221 			if(classLoader instanceof MainClassLoader && extraClassPath != null){
222 			    ((MainClassLoader) classLoader ).addClassPath(extraClassPath);			    
223 			}
224 			
225 			
226 			
227 
228 		} catch (Exception e) {
229 			e.printStackTrace();
230 			System.exit(0);
231 		}
232 		try {
233 			// set Font size offest for Xlets
234 			String s = Settings.getProperty("font.sizeoffset");
235 			xjava.awt.Font.setOffset(Integer.parseInt(s));
236 		} catch (NumberFormatException e) {
237 			e.printStackTrace();
238 		}
239 	}
