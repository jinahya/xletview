3   public class ArraySorter {
4   	/*
5   	 * * Sort in the same array
6   	 */
7   	public static void sort(Object[] a, Comparer comparer) {
8   		sort(a, null, 0, a.length - 1, true, comparer);
9   	}
10  
11  	/*
12  	 * * Sort a and b, using a as the reference
13  	 */
14  	public static void sort(Object[] a, Object[] b, int from, int to, boolean ascending, Comparer comparer) {
15  		// No sort
16  		if (a == null || a.length < 2)
17  			return;
18  
19  		// sort using Quicksort
20  		int i = from, j = to;
21  		Object center = a[(from + to) / 2];
22  		do {
23  			if (ascending) {
24  				while ((i < to) && (comparer.compare(center, a[i]) > 0))
25  					i++;
26  				while ((j > from) && (comparer.compare(center, a[j]) < 0))
27  					j--;
28  			} else {
29  				// Decending sort
30  				while ((i < to) && (comparer.compare(center, a[i]) < 0))
31  					i++;
32  				while ((j > from) && (comparer.compare(center, a[j]) > 0))
33  					j--;
34  			}
35  			if (i < j) {
36  				// Swap elements
37  				Object temp = a[i];
38  				a[i] = a[j];
39  				a[j] = temp;
40  				// Swap in b array if needed
41  				if (b != null) {
42  					temp = b[i];
43  					b[i] = b[j];
44  					b[j] = temp;
45  				}
46  			}
47  			if (i <= j) {
48  				i++;
49  				j--;
50  			}
51  		}
52  		while (i <= j);
53  		// Sort the rest
54  		if (from < j)
55  			sort(a, b, from, j, ascending, comparer);
56  		if (i < to)
57  			sort(a, b, i, to, ascending, comparer);
58  	}
59  }
