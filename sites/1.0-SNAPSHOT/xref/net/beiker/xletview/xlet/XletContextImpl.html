24  public class XletContextImpl implements XletContext{
25  	
26  	/**
27  	 * Debugging
28  	 */	
29  	private static final net.beiker.cake.Logger log = net.beiker.cake.Log.getLogger(XletContextImpl.class);
30  	
31  	// javaTV
32  	public static final String CONTAINER = "javax.tv.xlet.container";
33  	public static final String SERVICE_CONTEXT = "javax.tv.xlet.service_context";
34  	
35  	// xletview specific
36  	
37  //	public static final String ROOT_CONTAINER = "javax.tv.xlet.root_container";
38  	public static final String APP_DIR         = "app.dir";
39  	
40  	// spec 11.7
41  	private static final String PROPERTY_ORG_ID = "dvb.org.id";
42  	private static final String PROPERTY_APP_ID = "dvb.app.id";
43  	private static final String PROPERTY_CALLER_PARAMS = "dvb.caller.parameters";
44  
45      public static final int INITIALIZED = 0;
46      public static final int ACTIVE     = 2;
47      public static final int PAUSED      = 3;
48      public static final int DESTROYED   = -1;
49      private int state;
50      
51      private XletManager manager;
52      private Xlet xlet;
53      private ClassLoader classLoader;
54      
55      private EmulatorFile applicationDirectory;
56      
57      /*
58       * The array of strings returned by XletContext.getXletProperty(XletContext.ARGS) shall be the array of
59       * strings de?ned in the DVB-J application descriptor (see 10.9.1 on page 95) in the same order as speci?ed in the
60       * signalling. Each entry in the loop of that descriptor shall be presented as one string in the array returned by this method,
61       * interpreted using the encoding as speci?ed in 10.4.8, "Text encoding in AIT" on page 80. Zero-length strings in the
62       * signalling shall be represented as the empty string.
63       */
64      private String[] xletArgs = new String[0];
65      
66      /* 
67       * The "dvb.caller.parameters" XletProperty contains the array of Strings that was passed into the AppProxy.
68       * start(String[]) method by the application that started this application. If this application was started by the
69       * system or by another application using the AppProxy.start() method without parameters, an empty array with
70       * length 0 is returned as the value of this XletProperty.
71       */
72      private String[] callerArgs = new String[0];
73  
74      public XletContextImpl(XletManager manager, Xlet xlet, EmulatorFile appDir){
75          this.manager = manager;
76          this.xlet    = xlet;              
77          this.applicationDirectory = appDir;
78      }
79      
80  
81  
82      /* (non-Javadoc)
83       * @see xjavax.tv.xlet.XletContext#getXletProperty(java.lang.String)
84       */
85      public Object getXletProperty(String key){
86          log.debug("getXletProperty(" + key + ")");
87          if(key.equals(XletContext.ARGS)){
88          	return xletArgs;        	
89          }
90          else if(key.equals(PROPERTY_CALLER_PARAMS)){
91          	return callerArgs;
92          }
93          else if(key.equals(CONTAINER)){
94              return ScreenContainer.getInstance().getXletContainer();
95          }
96          else if(key.equals(SERVICE_CONTEXT)){
97              return ServiceContextImpl.getInstance();
98          }
99          else if(key.equals(APP_DIR)){
100         	return applicationDirectory;
101         }
102         else if(key.equals(PROPERTY_APP_ID)){
103         	log.debug("NOT IMPLEMENTED - RETURNING '0' AS APP_ID STRING.");
104         	return "0";
105         }
106         else if(key.equals(PROPERTY_ORG_ID)){
107         	log.debug("NOT IMPLEMENTED - RETURNING '0' AS ORG_ID STRING.");
108         	return "0";
109         }
110         else{
111         	log.debug("getXletProperty() for a key '"+key+"' that is unknown. Returning null.");
112             return null;
113         }
114     }

126      */
127     public void notifyPaused(){

129         manager.notifyPaused(this);
130     }

132     /* (non-Javadoc)
133      * @see xjavax.tv.xlet.XletContext#resumeRequest()
134      */
135     public void resumeRequest(){
136         manager.resumeRequest(this);
137     }

140         return state;
141     }
142

144         this.state = state;
145     }

148     	log.debug("xlet=" + xlet);
149         return xlet;
150     }
