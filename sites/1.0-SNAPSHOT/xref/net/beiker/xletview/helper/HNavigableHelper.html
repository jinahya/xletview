55  	/* (non-Javadoc)
56  	 * @see org.havi.ui.HNavigable#setMove(int, org.havi.ui.HNavigable)
57  	 */
58  	public void setMove(int keyCode, HNavigable target){
59  		if(navTargets == null){
60  			// create it
61  			navTargets = new Hashtable();
62  		}
63  		Integer code = new Integer(keyCode);
64  		// check if key already exist, if so, remove it
65  		if(navTargets.containsKey(code)){
66  			navTargets.remove(code);
67  		}
68  		// put the target, but only if target != null
69  		if(target != null){
70  			navTargets.put(code, target);
71  		}
72  	}

83  			HNavigable nav = (HNavigable) navTargets.get(code);

170 	/* (non-Javadoc)
171 	 * @see org.havi.ui.HNavigationInputPreferred#getNavigationKeys()
172 	 */
173 	public int[] getNavigationKeys(){
174 		if(navTargets == null){
175 			return null;
176 		}
177 		else if(navTargets.size() == 0){
178 			return null;
179 		}
180 		else{
181 			int[] keyCodes = new int[navTargets.size()];
182 			Enumeration enum_ = navTargets.elements();
183 			int i = 0;
184 			while(enum_.hasMoreElements()){
185 				Integer integer = (Integer) enum_.nextElement();
186 				keyCodes[i] = integer.intValue();
187 				i++;
188 			}
189 			return keyCodes;
190 		}
191 	}

193 	/**
194 	 * 
195 	 * @param evt
196 	 * @return The new state
197 	 */
198 	public int getHFocusEventResult(HFocusEvent evt) {
199 
200 		// get state
201 		int state = hVisible.getInteractionState(); 
202 		
203 		if(evt.getID() == FocusEvent.FOCUS_GAINED){
204 			
205 			// bitwise OR
206 			state = state | HState.FOCUSED_STATE_BIT;
207 			
208 			if(hFocusListener != null){
209 				hFocusListener.focusGained(evt);
210 			}
211 		}
212 		else if(evt.getID() == FocusEvent.FOCUS_LOST){            
213 			
214 			// bitwise XOR
215 			state = state ^ HState.FOCUSED_STATE_BIT;
216 			
217 			if(hFocusListener != null){
218 				hFocusListener.focusLost(evt);
219 			}
220 		}
221 		else if(evt.getID() == HFocusEvent.FOCUS_TRANSFER && evt.getTransferId() != HFocusEvent.NO_TRANSFER_ID){
222 			Integer navTarget = new Integer(evt.getTransferId());
223 			HNavigable newNav = (HNavigable)navTargets.get(navTarget);
224 			
225 			if(newNav instanceof Component){
226 				((Component)newNav).requestFocus();
227 				log.debug("requested focus on newNav");
228 			}    		
229 		}
230 		
231 		return state;
232 	}
