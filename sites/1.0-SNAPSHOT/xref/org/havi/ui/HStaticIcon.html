15  package org.havi.ui;
16  
17  import java.awt.Image;
18  
19  
20  /**
21   * 
22   * 
23   * @author Martin Sveden
24   * @statuscode 4
25   */
26  public class HStaticIcon extends HVisible implements HNoInputPreferred{
27  
28      private static HGraphicLook defaultHLook = new HGraphicLook();
29  
30      public HStaticIcon(){
31      	this(null, 0, 0, 0, 0);
32      }
33  
34      public HStaticIcon(Image imageNormal, int x, int y, int width, int height){
35          super(defaultHLook, x, y, width, height);        
36      	this.setGraphicContent(imageNormal, HVisible.NORMAL_STATE);
37      }
38  
39      public HStaticIcon(Image imageNormal){
40      	this(null, 0, 0, 0, 0);
41          this.setGraphicContent(imageNormal, HVisible.NORMAL_STATE);
42      }
43  
44      public void setLook(HLook hLook) throws HInvalidLookException{
45      	if(hLook instanceof HGraphicLook || hLook == null ){
46      		super.setLook(hLook);
47      	}
48      	else{
49      		throw new HInvalidLookException("HLook was not a org.havi.ui.HGraphicLook");
50      	}    	
51      }
52  
53      public static void setDefaultLook(HGraphicLook hGraphicLook){
54      	HStaticIcon.defaultHLook = hGraphicLook;
55      }
56  
57      public static HGraphicLook getDefaultLook(){
58          return HStaticIcon.defaultHLook;
59      }
60      
61      public String toString(){
62      	return super.toString() + " img:" + getGraphicContent(getInteractionState());
63      }
64  }
