23  /**
24   * A filter that only accepts applications that are signalled in the current service.
25   * @author Martin Sveden
26   * @statuscode 2
27   * @comment cannot be completed yet(dependencies)
28   */
29  public class CurrentServiceFilter extends AppsDatabaseFilter {
30  
31      private static final net.beiker.cake.Logger log = net.beiker.cake.Log.getLogger(CurrentServiceFilter.class);
32      
33  	public CurrentServiceFilter() {
34  		super();
35  	}
36  
37  	public boolean accept(AppID appId) { 
38  	    
39  	    // get the AppsDatabase
40  	    AppsDatabase db = AppsDatabase.getAppsDatabase();
41  	    
42  	    // get the Service Locator of the app
43  	    Locator appServiceLocator = db.getAppAttributes(appId).getServiceLocator();
44  	    
45  	    // get the Locator of the current Service
46  	    Locator currentServiceLocator = ServiceManager.getInstance().getCurrentService().getLocator();
47  	    
48  	    // check if the app is in the current service
49  	    try{
50  		    //TODO this will throw a nullpointer, implement the rest to make it work
51  		    currentServiceLocator.equals(appServiceLocator);
52  	    }
53  	    catch(NullPointerException e){
54  	        log.debug("accept(AppID) not implemented");
55  	    }
56  	    
57  		return false;
58  	}
59  
60  }
