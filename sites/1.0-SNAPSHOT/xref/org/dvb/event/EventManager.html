58  	public boolean addUserEventListener(UserEventListener listener, ResourceClient client, UserEventRepository userEvents)
59      throws IllegalArgumentException{ 
60          if(client == null){
61          	throw new IllegalArgumentException("ResourceClient was null");
62          }
63  	
64          listenerObjects.add(new ListenerItem(listener, userEvents.getUserEvent(), client));
65          fireResourceStatusEvent();
66          return true;
67      }
68     
69      public void addUserEventListener(UserEventListener listener, UserEventRepository userEvents) {
70          listenerObjects.add(new ListenerItem(listener, userEvents.getUserEvent()));
71      }

73      /**
74       * Removes all listeners. Not in the spec but used
75       * by XleTView to not keep listeners from destroyed
76       * Xlets. Quick and dirty solution for now.
77       *
78       */
79      public void removeAllUserEventListeners(){
80      	listenerObjects.clear();
81      }

83      public boolean addExclusiveAccessToAWTEvent (ResourceClient client, UserEventRepository userEvents)  
84      throws java.lang.IllegalArgumentException{
85          if(client == null){
86              throw new IllegalArgumentException("ResourceClient was null");            
87          }
88          
89          awtListenerObjects.add(new AwtListenerItem(client, userEvents.getUserEvent()));
90          
91          fireResourceStatusEvent();
92  
93          return true;
94      }

96      public void removeUserEventListener(UserEventListener listener){
97          for(int i = 0; i <listenerObjects.size(); i++){
98              ListenerItem li = (ListenerItem) listenerObjects.get(i);
99              UserEventListener ul = li.getListener();
100             if(ul == listener){
101                 listenerObjects.remove(i);
102             }
103         }
104     }
105    
106     public void removeExclusiveAccessToAWTEvent(ResourceClient client) {
107         for(int i = 0; i < awtListenerObjects.size(); i++){
108             AwtListenerItem li = (AwtListenerItem) listenerObjects.get(i);
109             ResourceClient rc = li.getResourceClient();
110             if(rc == client){
111                 awtListenerObjects.remove(i);
112                 fireResourceStatusEvent();
113             }
114         }   	   
115     }

117     public void addResourceStatusEventListener(ResourceStatusListener listener) {
118     	if(!resourceStatusListeners.contains(listener)){
119 			resourceStatusListeners.add(listener);		
120     	}
121     }
122 
123     public void removeResourceStatusEventListener(ResourceStatusListener listener) {
124 		resourceStatusListeners.remove(listener);

129     /**
130      * All key events are passed to this method that fires them 
131      * @param source
132      * @param keyEvent
133      */
134     public void fireUserEvent(Object source, KeyEvent keyEvent){
135         Component focusOwner = null;
136         int keyCode = keyEvent.getKeyCode(); 
137         char keyChar = keyEvent.getKeyChar();
138 
139         // user event
140         for(int i = listenerObjects.size()-1; i > -1; i--){
141         	ListenerItem li = (ListenerItem) listenerObjects.get(i);
142         	UserEvent[] userEvents = li.getEvents();
143         	for(int k = 0; k < userEvents.length; k++){
144         		if(userEvents[k].getCode() == keyCode){
145         			UserEvent ue = new UserEvent(source, UserEvent.UEF_KEY_EVENT, keyEvent.getID(), keyCode, -1, System.currentTimeMillis());
146         			li.getListener().userEventReceived(ue);
147         		}
148         	}
149         }
150         
151         HScene scene = XletManager.getInstance().getScene();        
152         
153         if(scene != null){
154             focusOwner = scene.getFocusOwner();
155         }
156         if(focusOwner != null){
157             //Debug.write(this, "focusOwner = " + focusOwner);
158             
159            
160             // awt event
161             KeyListener[] kl = focusOwner.getKeyListeners();
162             for(int i = 0; i < kl.length; i++){
163                 if(keyEvent.getID() == KeyEvent.KEY_PRESSED){
164                     kl[i].keyPressed(new KeyEvent(focusOwner,KeyEvent.KEY_PRESSED, 0L, 0, keyCode, keyChar));
165                 }
166                 else if(keyEvent.getID() == KeyEvent.KEY_RELEASED){
167                     kl[i].keyReleased(new KeyEvent(focusOwner,KeyEvent.KEY_RELEASED, 0L, 0, keyCode, keyChar));
168                 }
169                  
170                  
171             }
172         }
173         else{
174             //Debug.write(this, "focus owner is null");
175         }
176     }

178     /* Fired when there is a change in resouce clients */
179 	private void fireResourceStatusEvent(){
180 		for(int i = 0; i < resourceStatusListeners.size(); i++){
181 			ResourceStatusListener listener = (ResourceStatusListener) resourceStatusListeners.get(i);
182 			listener.statusChanged(new ResourceStatusEvent(this));
183 		}
184 	}

187     /*
188      * Class that makes it easier to handle the listener/events-from-repository
189      * 
190      * */
191     private class ListenerItem{
192         private UserEventListener listener;
193         private UserEvent[] events;
194         private ResourceClient client;
195 
196         private ListenerItem(UserEventListener listener, UserEvent[] events, ResourceClient client){
197             this(listener, events);
198             this.client = client;
199         }
200 
201         private ListenerItem(UserEventListener listener, UserEvent[] events){
202             this.listener = listener;
203             this.events = events;
204 //            Debug.write(this, "events.length = " + events.length);
205         }
206         
207         private UserEventListener getListener(){
208             return listener;
209         }
210         
211         private UserEvent[] getEvents(){
212             return events;
213         }    
214         
215         private ResourceClient getClient(){
216             return client;
217         }    
218     }
219     
220     private class AwtListenerItem{
221         private ResourceClient resourceClient;
222         private UserEvent[] events;
223         
224         private AwtListenerItem(ResourceClient resourceClient, UserEvent[] events){
225             this.resourceClient = resourceClient;
226             this.events = events;
227         }   
228         
229         private ResourceClient getResourceClient(){
230             return resourceClient;
231         }
232         
233         private UserEvent[] getEvents(){
234             return events;
235         }
236     }
